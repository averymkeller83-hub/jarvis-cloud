      var SAFE_TAGS = { STRONG: 1, EM: 1, CODE: 1, PRE: 1, A: 1, BR: 1, P: 1, UL: 1, OL: 1, LI: 1 };

      function sanitizeNode(parent) {
        var children = Array.from(parent.childNodes);
        for (var i = 0; i < children.length; i++) {
          var child = children[i];
          if (child.nodeType === 1) {
            if (!SAFE_TAGS[child.tagName]) {
              var text = document.createTextNode(child.textContent || '');
              parent.replaceChild(text, child);
            } else {
              var attrs = Array.from(child.attributes);
              for (var j = 0; j < attrs.length; j++) {
                var name = attrs[j].name;
                if (child.tagName === 'A' && (name === 'href' || name === 'target' || name === 'rel')) {
                  var href = child.getAttribute('href') || '';
                  if (name === 'href' && href.indexOf('http') !== 0) {
                    child.removeAttribute('href');
                  }
                  continue;
                }
                child.removeAttribute(name);
              }
              if (child.tagName === 'A') {
                child.setAttribute('target', '_blank');
                child.setAttribute('rel', 'noopener noreferrer');
              }
              sanitizeNode(child);
            }
          }
        }
      }

      function renderMarkdown(text) {
        var escaped = text
          .replace(/&/g, '&amp;')
          .replace(/</g, '&lt;')
          .replace(/>/g, '&gt;')
          .replace(/"/g, '&quot;');

        var html = escaped
          .replace(/```([\s\S]*?)```/g, '<pre><code>$1</code></pre>')
          .replace(/`([^`]+)`/g, '<code>$1</code>')
          .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')
          .replace(/\*(.+?)\*/g, '<em>$1</em>')
          .replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2">$1</a>')
          .replace(/(?:^|\n)[-•] (.+)/g, '<li>$1</li>')
          .replace(/(?:^|\n)(\d+)\. (.+)/g, '<li>$2</li>')
          .replace(/(<li>.*<\/li>)/g, '<ul>$1</ul>')
          .replace(/<\/ul>\s*<ul>/g, '')
          .replace(/\n\n/g, '</p><p>')
          .replace(/\n/g, '<br>');
        html = '<p>' + html + '</p>';
        html = html.replace(/<p><\/p>/g, '');

        var container = document.createElement('div');
        container.innerHTML = html;
        sanitizeNode(container);
        return container;
      }

      function b64urlToUint8(base64String) {
        var padding = '='.repeat((4 - base64String.length % 4) % 4);
        var base64 = (base64String + padding).replace(/-/g, '+').replace(/_/g, '/');
        var rawData = atob(base64);
        var arr = new Uint8Array(rawData.length);
        for (var i = 0; i < rawData.length; i++) arr[i] = rawData.charCodeAt(i);
        return arr;
      }

      function connect() {
        if (ws) { try { ws.close(); } catch(e) {} }

        var proto = location.protocol === 'https:' ? 'wss:' : 'ws:';
        var url = proto + '//' + location.host + '/ws/chat';

        updateStatus('connecting');

        try {
          ws = new WebSocket(url);
        } catch(e) {
          updateStatus('error');
          scheduleReconnect();
          return;
        }

        ws.onopen = function() {
          reconnectDelay = 1000;
          reconnectAttempts = 0;
          updateStatus('connected');
          if (keepaliveTimer) clearInterval(keepaliveTimer);
          keepaliveTimer = setInterval(function() {
            if (ws && ws.readyState === 1) {
              try { ws.send(JSON.stringify({ type: 'ping' })); } catch(e) {}
            }
          }, 30000);
        };

        ws.onmessage = function(evt) {
          try { handleMessage(JSON.parse(evt.data)); } catch(e) {}
        };

        ws.onclose = function(evt) {
          ws = null;
          if (keepaliveTimer) { clearInterval(keepaliveTimer); keepaliveTimer = null; }
          if (evt.code === 4001 || evt.code === 1008) {
            window.location.href = '/login';
            return;
          }
          updateStatus('disconnected');
          scheduleReconnect();
        };

        ws.onerror = function() {};
      }

      function scheduleReconnect() {
        if (reconnectTimer) return;
        reconnectAttempts++;
        if (reconnectAttempts >= 15) {
          window.location.href = '/login';
          return;
        }
        reconnectTimer = setTimeout(function() {
          reconnectTimer = null;
          reconnectDelay = Math.min(reconnectDelay * 1.5, 15000);
          connect();
        }, reconnectDelay);
      }

      document.addEventListener('visibilitychange', function() {
        if (!document.hidden && (!ws || ws.readyState !== 1)) {
          reconnectDelay = 1000;
          connect();
        }
      });

      function chunkText(text, maxLen) {
        var chunks = [];
        while (text.length > 0) {
          if (text.length <= maxLen) { chunks.push(text); break; }
          var cut = -1;
          for (var i = maxLen; i > maxLen * 0.5; i--) {
            var ch = text[i];
            if (ch === '.' || ch === '!' || ch === '?' || ch === '\n') { cut = i + 1; break; }
          }
          if (cut === -1) {
            for (var j = maxLen; j > maxLen * 0.5; j--) {
              if (text[j] === ' ' || text[j] === ',') { cut = j + 1; break; }
            }
          }
          if (cut === -1) cut = maxLen;
          chunks.push(text.substring(0, cut).trim());
          text = text.substring(cut).trim();
        }
        return chunks;
      }

      var PAUSE_SVG = '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><rect x="6" y="4" width="4" height="16"/><rect x="14" y="4" width="4" height="16"/></svg>';

      function createTtsBtn(text) {
        var btn = document.createElement('button');
        btn.className = 'tts-btn';
        btn.innerHTML = PLAY_SVG;
        btn.addEventListener('click', function() {
          btn.classList.remove('pulse');
          if (btn.classList.contains('playing')) {
            ttsPlayer.pause();
            ttsPlayer.currentTime = 0;
            if (ttsAbort) { ttsAbort.cancelled = true; ttsAbort = null; }
            btn.classList.remove('playing');
            btn.innerHTML = PLAY_SVG;
            setOrbState('idle');
            return;
          }
          var clean = text.replace(/```[\s\S]*?```/g, '').replace(/[#*_~`>]/g, '').trim();
          if (!clean) return;
          document.querySelectorAll('.tts-btn.playing').forEach(function(b) {
            b.classList.remove('playing');
            b.innerHTML = PLAY_SVG;
          });
          btn.classList.add('playing');
          btn.innerHTML = PAUSE_SVG;
          setOrbState('speaking');
          var resetBtn = function() {
            btn.classList.remove('playing');
            btn.innerHTML = PLAY_SVG;
            setOrbState('idle');
          };
          var ctx = new (window.AudioContext || window.webkitAudioContext)();
          var buf = ctx.createBuffer(1, 1, 22050);
          var src = ctx.createBufferSource();
          src.buffer = buf;
          src.connect(ctx.destination);
          src.start(0);
          var chunks = chunkText(clean, 900);
          var session = { cancelled: false };
          ttsAbort = session;
          var voice = getVoice();
          Promise.all(chunks.map(function(chunk) {
            return fetch('/api/tts', {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              credentials: 'same-origin',
              body: JSON.stringify({ text: chunk, voice: voice }),
            }).then(function(r) {
              if (!r.ok) return r.json().then(function(d) { throw new Error(d.error || 'TTS error'); });
              return r.blob();
            });
          })).then(function(blobs) {
            if (session.cancelled) return;
            var combined = new Blob(blobs, { type: 'audio/mpeg' });
            if (ttsObjUrl) URL.revokeObjectURL(ttsObjUrl);
            ttsObjUrl = URL.createObjectURL(combined);
            ttsPlayer.src = ttsObjUrl;
            ttsPlayer.onended = function() { resetBtn(); ctx.close(); };
            ttsPlayer.onerror = function() { resetBtn(); ctx.close(); };
            return ttsPlayer.play();
          }).catch(function(e) {
            console.error('TTS failed:', e);
            showVoiceToast(e.message || 'Voice unavailable');
            resetBtn();
            ctx.close();
          });
        });
        return btn;
      }

      function processFile(file) {
        if (file.size > MAX_FILE_SIZE) {
          addSystem('File too large: ' + file.name + ' (max 20MB)');
          return;
        }
        if (file.type.startsWith('image/')) {
          resizeAndAttach(file);
        } else {
          readFileAsText(file);
        }
      }

      function resizeAndAttach(file) {
        var reader = new FileReader();
        reader.onload = function(e) {
          var img = new Image();
          img.onload = function() {
            var w = img.width, h = img.height;
            if (w > MAX_IMAGE_DIM || h > MAX_IMAGE_DIM) {
              var scale = MAX_IMAGE_DIM / Math.max(w, h);
              w = Math.round(w * scale);
              h = Math.round(h * scale);
            }
            var canvas = document.createElement('canvas');
            canvas.width = w;
            canvas.height = h;
            var ctx = canvas.getContext('2d');
            ctx.drawImage(img, 0, 0, w, h);
            var dataUrl = canvas.toDataURL('image/jpeg', 0.85);
            var b64 = dataUrl.split(',')[1];
            var attachment = { type: 'image', mime: 'image/jpeg', data: b64, name: file.name, preview: dataUrl };
            pendingAttachments.push(attachment);
            renderAttachPreview();
          };
          img.src = e.target.result;
        };
        reader.readAsDataURL(file);
      }

      function readFileAsText(file) {
        var reader = new FileReader();
        reader.onload = function(e) {
          var text = e.target.result;
          if (text.length > MAX_FILE_SIZE) {
            addSystem('File content too large: ' + file.name);
            return;
          }
          var attachment = { type: 'file', mime: file.type || 'text/plain', data: btoa(unescape(encodeURIComponent(text))), name: file.name };
          pendingAttachments.push(attachment);
          renderAttachPreview();
        };
        reader.readAsText(file);
      }

      function send() {
        var text = input.value.trim();
        var hasAttachments = pendingAttachments.length > 0;
        if ((!text && !hasAttachments) || !ws || ws.readyState !== WebSocket.OPEN) return;
        voiceConversationActive = false;

        if (text && !hasAttachments) {
          var lower = text.toLowerCase();
          if (lower[0] === '/' || lower[0] === '\\') {
            var parts = lower.slice(1).split(' ');
            var cmdKey = parts[0];
            var systemCmd = SLASH_COMMANDS[cmdKey];
            addUser(text);
            if (systemCmd) {
              if (systemCmd === 'clear') {
                localStorage.removeItem(CHAT_STORE_KEY);
                clearPending();
                while (messagesEl.firstChild) messagesEl.removeChild(messagesEl.firstChild);
              }
              var cmdArgs = parts.slice(1).join(' ');
              var fullCmdStr = cmdArgs ? systemCmd + ' ' + cmdArgs : systemCmd;
              ws.send(JSON.stringify({ type: 'command', command: fullCmdStr }));
            } else {
              saveChatMsg('user', text);
              ws.send(JSON.stringify({ type: 'command', command: text.slice(1) }));
            }
            setPending('cmd');
            showThinking();
            input.value = '';
            input.style.height = 'auto';
            input.focus();
            return;
          }
        }

        var displayText = text || '';
        var attachHtml = '';
        var attachments = [];
        for (var i = 0; i < pendingAttachments.length; i++) {
          var a = pendingAttachments[i];
          attachments.push({ type: a.type, mime: a.mime, data: a.data, name: a.name });

            attachHtml += '<img class="msg-image" src="' + a.preview + '">';

            attachHtml += '<div class="msg-file-badge"><span class="file-icon">📄</span>' + a.name + '</div>';
          }

        addUser(displayText + (attachHtml ? '<br>' + attachHtml : ''), true);
        saveChatMsg('user', displayText || '[attachment]');

        var id = crypto.randomUUID();
        var payload = { type: 'chat', id: id, message: text || '' };
        if (attachments.length > 0) payload.attachments = attachments;
        ws.send(JSON.stringify(payload));
        setPending(id);
        showThinking();

        pendingAttachments = [];
        renderAttachPreview();
        input.value = '';
        input.style.height = 'auto';
        input.focus();
      }
